-- Create a table for users (students and staff)
CREATE TABLE Users (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    password VARCHAR(100) NOT NULL,
    role VARCHAR(10) CHECK (role IN ('student', 'staff')) NOT NULL
);

-- Create a table for cafes and bars
CREATE TABLE CafesBars (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    location VARCHAR(100),
    opening_hours VARCHAR(100)
);

-- Create a table for food items
CREATE TABLE FoodItems (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    description TEXT,
    price DECIMAL(5, 2) NOT NULL,
    cafe_bar_id INT REFERENCES CafesBars(id) ON DELETE CASCADE
);

-- Create a table for menus
CREATE TABLE Menus (
    id SERIAL PRIMARY KEY,
    cafe_bar_id INT REFERENCES CafesBars(id) ON DELETE CASCADE,
    date DATE NOT NULL,
    food_item_id INT REFERENCES FoodItems(id) ON DELETE CASCADE
);

-- Create a table for orders
CREATE TABLE Orders (
    id SERIAL PRIMARY KEY,
    user_id INT REFERENCES Users(id) ON DELETE CASCADE,
    order_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    status VARCHAR(20) CHECK (status IN ('pending', 'completed', 'cancelled')) DEFAULT 'pending',
    total_price DECIMAL(10, 2) NOT NULL
);

-- Create a table for order items
CREATE TABLE OrderItems (
    id SERIAL PRIMARY KEY,
    order_id INT REFERENCES Orders(id) ON DELETE CASCADE,
    food_item_id INT REFERENCES FoodItems(id) ON DELETE CASCADE,
    quantity INT NOT NULL,
    price DECIMAL(5, 2) NOT NULL
);

-- Create a table for scheduled takeaways
CREATE TABLE ScheduledTakeaways (
    id SERIAL PRIMARY KEY,
    user_id INT REFERENCES Users(id) ON DELETE CASCADE,
    order_id INT REFERENCES Orders(id) ON DELETE CASCADE,
    scheduled_time TIMESTAMP NOT NULL
);

-- Create a table for payments
CREATE TABLE Payments (
    id SERIAL PRIMARY KEY,
    order_id INT REFERENCES Orders(id) ON DELETE CASCADE,
    payment_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    amount DECIMAL(10, 2) NOT NULL,
    payment_method VARCHAR(20) CHECK (payment_method IN ('credit_card', 'debit_card', 'paypal', 'cash')) NOT NULL
);
